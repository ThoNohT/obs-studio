project(automation)

if(WIN32)
	set(obs-outputs_PLATFORM_DEPS
		ws2_32
		winmm)
endif()

if(MSVC)
	set(obs-outputs_PLATFORM_DEPS
		${obs-outputs_PLATFORM_DEPS}
		w32-pthreads)
endif()

set(automation_SOURCES
	automation.c)

add_library(automation MODULE
	${automation_SOURCES})

target_link_libraries(automation
		libobs
		${obs-outputs_PLATFORM_DEPS})

install_obs_plugin_with_data(automation data)
